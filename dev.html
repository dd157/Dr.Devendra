function animateCounters(){

  document.querySelectorAll('.stat-num,.metric-num').forEach(el=>{

    const text = el.textContent;

    const num = parseFloat(text.replace(/[^0-9.]/g,''));

    if(!num) return;

    const prefix = text.match(/[₹]/)?.[0] || '';

    const suffix = text.match(/[+L]/)?.[0] || '';

    let current = 0; const steps = 40;

    const inc = num/steps;

    const t = setInterval(()=>{

      current = Math.min(current+inc, num);

      el.textContent = prefix + (Number.isInteger(num)?Math.round(current):current.toFixed(1)) + suffix;

      if(current>=num) clearInterval(t);

    }, 40);

  });

}
